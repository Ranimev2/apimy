Dokumentasi API Yanz-GPT


const axios = require('axios');            	

exports.YanzGPT = (query) => {
  return new Promise(async (resolve, reject) => {
    try {
      const response = await axios("https://yotsubaai.vercel.app", {
        data: {
          query: query
        },
        headers: {
          "Content-Type": "application/json"
        },
        method: "POST"
      });
      resolve(response.data.message);
    } catch (error) {
      reject(error);
    }
  });
};
